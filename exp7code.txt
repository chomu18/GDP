# Experiment No. 7A: Control of LED using Raspberry Pi GPIO
# Import necessary libraries
import RPi.GPIO as GPIO
from time import sleep
# Set the GPIO mode (BCM numbering)
GPIO.setmode(GPIO.BCM)
# Define LED pin numbers
led_pins = [17, 27, 22] # GPIO pins connected to LEDs
# Set all LED pins as output
for pin in led_pins:
GPIO.setup(pin, GPIO.OUT)
GPIO.output(pin, GPIO.LOW)
print("Experiment 7A: LED Blinking using Raspberry Pi")
print("Press Ctrl+C to stop the program.")
try:
while True:
# Turn ON all LEDs
for pin in led_pins:
GPIO.output(pin, GPIO.HIGH)
print("LEDs ON")
sleep(1)
# Turn OFF all LEDs
for pin in led_pins:
GPIO.output(pin, GPIO.LOW)
print("LEDs OFF")
sleep(1)
except KeyboardInterrupt:
print("\nProgram stopped by user.")
GPIO.cleanup()
